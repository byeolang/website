8 // Implementing Macro Overloding:
9 // by count of given arguments, let compiler determine which macro should works.
10 // original source code from BuvinJ at
11 // https://stackoverflow.com/questions/11761703/overloading-macro-on-number-of-arguments
12 //
13 // usage:
14 // // define macro:
15 // #define MyMacro_0() 100
16 // #define MyMacro_1(x) (x)+5
17 // #define MyMacro_2(x, y) (x)+(y)
18 // #define MyMacro(...) BY_OVERLOAD(MyMacro, __VA_ARGS__)
19 //
20 // // using:
21 // cout << MyMacro(5) << ", " << MyMacro(3, 5) << ", " << MyMacro() << "\n";
22 //
23 // output:
24 // 10, 8, 100
25 #define BY_OVERLOAD(NAME, ...) BY_CONCAT(NAME##_, __OVERLOAD_VA_NUM_ARGS(__VA_ARGS__))(__VA_ARGS__)
